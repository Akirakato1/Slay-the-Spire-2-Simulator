// Potion-effect data table — batch_p_2
//
// Re-encoding pass after Effect VM vocabulary expansion. New primitives now
// available (vs. batch_p_1):
//   * Effect::AddRandomCardFromPool { pool, filter, n, pile, upgrade,
//       free_this_turn, distinct }  — unlocks the random-card-from-pool
//       factories (AttackPotion / SkillPotion / PowerPotion /
//       ColorlessPotion / CosmicConcoction / OrobicAcid).
//   * Effect::SummonOsty { osty_id, max_hp: Option<AmountSpec> }  — adds
//       the HP arg that was missing in batch_p_1; unlocks BoneBrew.
//   * Effect::HealOsty { amount }.
//   * AmountSpec::TargetMaxHp  — unlocks target-MaxHp-relative HEAL effects.
//   * AmountSpec::TargetBlock  — unlocks Fortifier (target.Block * 2).
//   * AmountSpec::DistinctOrbTypesInQueue.
//
// Conventions inherited from batch_p_1 (same canonical var keys,
// Unpowered flag dropped, custom-power literal type names, etc).
//
// CunningPotion / PotOfGhouls are *not* pool-factories — they spawn a
// fixed card-id N times (Shiv.CreateInHand / Soul.CreateInHand). Encoded
// as Repeat { count: Canonical("Cards"), body: [AddCardToPile {...}] },
// matching the pattern already used by FuneraryMask (relic emitting Soul
// each turn) in effects.rs. CunningPotion applies CardCmd.Upgrade to
// each spawned Shiv → upgrade: 1.
//
// Still-SKIPped potions and the primitive gap each one still needs are
// listed in the §Summary at the bottom.

// ---------- ENCODED ARMS ----------

// AttackPotion.cs L48-62: TargetType.Self, CombatOnly.
// CardFactory.GetDistinctForCombat(owner.Character.CardPool unlocked
// where Type==Attack, 3, rng) → modal choose 1 → SetToFreeThisTurn →
// AddGeneratedCardToCombat(Hand, Bottom). Collapsing the choose-1-of-3
// modal to "give all 3 free-this-turn" matches the encoding shape
// endorsed in the AddRandomCardFromPool docstring on effects.rs L487.
"AttackPotion" => Some(vec![Effect::AddRandomCardFromPool {
    pool: CardPoolRef::CharacterAttack,
    filter: CardFilter::Any,
    n: AmountSpec::Fixed(3),
    pile: Pile::Hand,
    upgrade: 0,
    free_this_turn: true,
    distinct: true,
}]),

// SkillPotion.cs L48-62: same shape as AttackPotion, Type==Skill.
"SkillPotion" => Some(vec![Effect::AddRandomCardFromPool {
    pool: CardPoolRef::CharacterSkill,
    filter: CardFilter::Any,
    n: AmountSpec::Fixed(3),
    pile: Pile::Hand,
    upgrade: 0,
    free_this_turn: true,
    distinct: true,
}]),

// PowerPotion.cs L48-62: same shape as AttackPotion, Type==Power.
"PowerPotion" => Some(vec![Effect::AddRandomCardFromPool {
    pool: CardPoolRef::CharacterPower,
    filter: CardFilter::Any,
    n: AmountSpec::Fixed(3),
    pile: Pile::Hand,
    upgrade: 0,
    free_this_turn: true,
    distinct: true,
}]),

// ColorlessPotion.cs L49-61: TargetType.Self, CombatOnly.
// CardFactory.GetDistinctForCombat over ModelDb.CardPool<ColorlessCardPool>
// .GetUnlockedCards, n=3, then modal choose 1 + SetToFreeThisTurn.
// Collapsed to "give all 3 free-this-turn" (same shape as AttackPotion).
"ColorlessPotion" => Some(vec![Effect::AddRandomCardFromPool {
    pool: CardPoolRef::Colorless,
    filter: CardFilter::Any,
    n: AmountSpec::Fixed(3),
    pile: Pile::Hand,
    upgrade: 0,
    free_this_turn: true,
    distinct: true,
}]),

// CosmicConcoction.cs L62-71: TargetType.Self, CombatOnly. CanonicalVar
// Cards=3. For each of N distinct colorless cards: CardCmd.Upgrade →
// AddGeneratedCardToCombat(Hand, Bottom). No modal choose, no
// SetToFreeThisTurn (cards are added at base cost). Encoded with
// upgrade: 1 (CardCmd.Upgrade fires before add) and
// free_this_turn: false. Count threads through the Cards canonical var.
"CosmicConcoction" => Some(vec![Effect::AddRandomCardFromPool {
    pool: CardPoolRef::Colorless,
    filter: CardFilter::Any,
    n: AmountSpec::Canonical("Cards".to_string()),
    pile: Pile::Hand,
    upgrade: 1,
    free_this_turn: false,
    distinct: true,
}]),

// CunningPotion.cs L72-81: TargetType.Self, CombatOnly. CanonicalVar
// Cards=3. Shiv.CreateInHand(owner, 3) → for each spawned Shiv,
// CardCmd.Upgrade. Shiv is a specific card, not a pool — encoded as
// Repeat { Cards × AddCardToPile(Shiv+, Hand) }. upgrade: 1 mirrors
// the per-card Upgrade. Matches FuneraryMask's Repeat-AddCardToPile shape.
"CunningPotion" => Some(vec![Effect::Repeat {
    count: AmountSpec::Canonical("Cards".to_string()),
    body: vec![Effect::AddCardToPile { card_id: "Shiv".to_string(), upgrade: 1, pile: Pile::Hand }],
}]),

// OrobicAcid.cs L48-67: TargetType.Self, CombatOnly. Three separate
// CardFactory.GetDistinctForCombat calls — 1 Attack, 1 Skill, 1 Power
// from owner's character pool — concatenated, each SetToFreeThisTurn,
// then AddGeneratedCardsToCombat(Hand, Bottom). Encoded as three
// back-to-back AddRandomCardFromPool calls. No CanonicalVars on this
// potion: counts are literal 1s.
"OrobicAcid" => Some(vec![
    Effect::AddRandomCardFromPool {
        pool: CardPoolRef::CharacterAttack,
        filter: CardFilter::Any,
        n: AmountSpec::Fixed(1),
        pile: Pile::Hand,
        upgrade: 0,
        free_this_turn: true,
        distinct: true,
    },
    Effect::AddRandomCardFromPool {
        pool: CardPoolRef::CharacterSkill,
        filter: CardFilter::Any,
        n: AmountSpec::Fixed(1),
        pile: Pile::Hand,
        upgrade: 0,
        free_this_turn: true,
        distinct: true,
    },
    Effect::AddRandomCardFromPool {
        pool: CardPoolRef::CharacterPower,
        filter: CardFilter::Any,
        n: AmountSpec::Fixed(1),
        pile: Pile::Hand,
        upgrade: 0,
        free_this_turn: true,
        distinct: true,
    },
]),

// PotOfGhouls.cs L70-74: TargetType.Self, CombatOnly. CanonicalVar
// Cards=2. Soul.CreateInHand(owner, 2). Soul is a specific card; no
// upgrade pass. Encoded as Repeat over AddCardToPile(Soul, Hand). Same
// shape as CunningPotion (and FuneraryMask).
"PotOfGhouls" => Some(vec![Effect::Repeat {
    count: AmountSpec::Canonical("Cards".to_string()),
    body: vec![Effect::AddCardToPile { card_id: "Soul".to_string(), upgrade: 0, pile: Pile::Hand }],
}]),

// BoneBrew.cs L72-81: TargetType.Self, CombatOnly. CanonicalVar
// Summon=15. OstyCmd.Summon(choiceContext, owner, Summon.BaseValue,
// this). New SummonOsty primitive carries Option<AmountSpec> max_hp →
// thread Summon canonical var through.
"BoneBrew" => Some(vec![Effect::SummonOsty {
    osty_id: "Default".to_string(),
    max_hp: Some(AmountSpec::Canonical("Summon".to_string())),
}]),

// Fortifier.cs L61-66: TargetType.AnyPlayer, CombatOnly.
// CreatureCmd.GainBlock(target, target.Block * 2, Unpowered). Both
// operands are literal: target.Block (now an AmountSpec) and the
// integer 2. Encoded as GainBlock(Mul(TargetBlock, Fixed(2))).
// Unpowered prop dropped (same as BlockPotion / ShipInABottle).
"Fortifier" => Some(vec![Effect::GainBlock {
    amount: AmountSpec::Mul {
        left: Box::new(AmountSpec::TargetBlock),
        right: Box::new(AmountSpec::Fixed(2)),
    },
    target: Target::SelfPlayer,
}]),

// ---------- STILL-SKIPped ----------

// SKIP BloodPotion: CreatureCmd.Heal(target, target.MaxHp * HealPercent
// / 100m) where DynamicVar "HealPercent" = 20m. Encoding Mul {
// TargetMaxHp, Canonical("HealPercent") } would give MaxHp * 20 (~20×
// over-heal), not MaxHp * 0.20. Needs either AmountSpec::Div or a
// percent-aware AmountSpec variant. Vocabulary gap, not encoding gap.

// SKIP FairyInABottle: two gaps. (1) Triggers from AfterPreventingDeath
// hook, not OnUse — Effect VM has no potion-side automatic-trigger
// hook table. (2) Heal amount is Math.Max(target.MaxHp * 0.3m, 1m) —
// decimal multiplier on TargetMaxHp; same Div/percent-aware AmountSpec
// gap as BloodPotion.

// SKIP EssenceOfDarkness: body is `for i in 0..OrbQueue.Capacity →
// OrbCmd.Channel<DarkOrb>`. Needs AmountSpec::EmptyOrbSlots (or
// OrbSlotCapacity) for the Repeat count. Encoding as Repeat { Fixed(3)
// } would bake the default capacity and silently drift if the player
// has PotionOfCapacity-extended slots — strictly worse than SKIP.

// SKIP FoulPotion: PassesCustomUsabilityCheck and OnUse both branch on
// RunState.CurrentRoom type (combat / MerchantRoom / FakeMerchant
// EventRoom), with cross-player NRun.EventSynchronizer dispatch in the
// last arm. Multi-branch on out-of-combat room state is outside the
// Effect VM's scope.

// SKIP GamblersBrew: CardSelectCmd.FromHandForDiscard (unbounded N) →
// CardCmd.DiscardAndDraw(list, list.Count). Discard-then-draw where
// the draw amount = the picked count needs a pick-result variable
// (AmountSpec::PickedCardCount) to thread between steps. Same gap
// noted in batch_p_1.

// SKIP SneckoOil: per-card RNG mutation. CardPileCmd.Draw(Cards) then
// for each non-X-cost hand card with cost≥0, set this-turn-or-until-
// played cost to RNG.NextInt(4). SetCardCost takes one fixed
// AmountSpec; encoding a per-card independent RNG sample is not
// possible without a CostSourceMode::RandomInt(max).

// SKIP SoldiersStew: increments BaseReplayCount on every Strike-tagged
// card in CombatState.AllCards. No primitive for per-card-instance
// numeric mutation (ApplyKeywordToCards is keyword-only, stub).

// SKIP TouchOfInsanity: needs SetCardCost(scope: ThisCombat) to
// actually mutate CardInstance cost. Primitive exists in vocab but the
// dispatcher arm is STUB (effects.rs L3176-3179). Keeping SKIP so the
// batch's encoded rows stay runnable.

// SKIP DeprecatedPotion: no OnUse override (deprecated stub). Same
// reason as batch_p_1.
