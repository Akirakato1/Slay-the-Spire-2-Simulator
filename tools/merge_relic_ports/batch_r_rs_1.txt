// ============================================================================
// batch_r_rs_1.txt — AfterObtained relic run-state encodings (batch 1)
//
// Bulk-encoded AfterObtained hooks for ~75 relics defined in
// sts2_decompiled/sts2/MegaCrit/sts2/Core/Models/Relics/*.cs that override
// AfterObtained(). The six already encoded in run_state_effects
// (Mango/Pear/Strawberry/FakeMango/OldCoin/CursedPearl) are not duplicated
// here.
//
// Each Some-arm uses Effect variants from the supported run-state set:
//   GainRunStateMaxHp / GainRunStateGold / LoseRunStateHp /
//   GainRelic / LoseRelic / GainPotionToBelt
// AmountSpec::Canonical resolves to 0 at run-state scope, so literal
// Fixed(N) is used where N is the C# DynamicVar BaseValue.
//
// SKIPs document the precise reason — usually because the C# body relies
// on a primitive the dispatcher doesn't yet implement (mid-run deck
// mutation, max-potion-count, transform/upgrade, etc.).
// ============================================================================

// ===== AfterObtained: permanent +MaxHp relics =====

// BigMushroom.cs L55-59: GainMaxHp(20) then `this.Grow()` (visual only).
// CanonicalVars: new MaxHpVar(20m), new CardsVar(2). The Cards var is for
// ModifyHandDraw (first-round draw -2); only the +MaxHp lands at pickup.
"BigMushroom" => Some(vec![(
    RunStateHook::AfterObtained,
    vec![Effect::GainRunStateMaxHp { amount: AmountSpec::Fixed(20) }],
)]),

// LeesWaffle.cs L48-53: GainMaxHp(7) then Heal to full.
// CanonicalVars: new MaxHpVar(7m). GainRunStateMaxHp also raises current
// HP by the gained amount (see execute_run_state_effect), which mirrors
// the "heal to full" effect for a freshly-bought waffle in practice
// (player is typically near full HP at shops anyway). The extra Heal
// edge case for damaged players is not perfectly modeled — flagging as
// a low-priority dispatcher gap (would need a HealToFull primitive).
"LeesWaffle" => Some(vec![(
    RunStateHook::AfterObtained,
    vec![Effect::GainRunStateMaxHp { amount: AmountSpec::Fixed(7) }],
)]),

// LoomingFruit.cs L73-77: GainMaxHp(31). CanonicalVars: new MaxHpVar(31m).

)]),

// NutritiousOyster.cs L47-50: GainMaxHp(11). CanonicalVars: new MaxHpVar(11m).
