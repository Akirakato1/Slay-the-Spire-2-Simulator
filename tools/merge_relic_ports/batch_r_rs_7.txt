// batch_r_rs_7.txt — interactive deck-pick relics. C# uses
// CardSelectCmd.FromDeckFor{Transformation,Enchantment,Upgrade,Removal}
// which prompts the player. Sim has no player UI, so we substitute the
// random variants of each primitive (UpgradeRandomDeckCards,
// EnchantRandomDeckCards, RemoveRandomDeckCards, etc.) using the
// up_front RNG. Net behavior: same NUMBER of cards affected, but the
// specific cards differ from a player's manual pick.

// Astrolabe.cs L51-63: transform 3 chosen deck cards.
"Astrolabe" => Some(vec![(
    RunStateHook::AfterObtained,
    vec![Effect::TransformRandomDeckCards {
        n: AmountSpec::Fixed(3),
        filter: CardFilter::Any,
        pool: CardPoolRef::CharacterAny,
    }],
)]),

// BeautifulBracelet.cs L57-77: enchant 3 chosen with Swift(3).
"BeautifulBracelet" => Some(vec![(
    RunStateHook::AfterObtained,
    vec![Effect::EnchantRandomDeckCards {
        n: AmountSpec::Fixed(3),
        filter: CardFilter::Any,
        enchantment_id: "Swift".to_string(),
        enchantment_amount: 3,
    }],
)]),

// BiiigHug: remove 4 chosen. AfterShuffle Soot-add side is combat-frame
// and SKIPped here.

    vec![Effect::RemoveRandomDeckCards {
        n: AmountSpec::Fixed(4),
        filter: CardFilter::Any,
    }],
)]),

// Claws.cs L53-75: transform 1 chosen into Maul (replace). The original
// is replaced (not removed + Maul added) — preserves upgrade level on
// upgrade-eligible target. Our ReplaceRandomDeckCard sets new id but
// drops upgrade/enchantment; close enough.
"Claws" => Some(vec![(
    RunStateHook::AfterObtained,
    vec![Effect::ReplaceRandomDeckCard {
        from_filter: CardFilter::Any,
        to_card_id: "Maul".to_string(),
    }],
)]),
